<div class="notranslate"><p>给定一个大小为&nbsp;<em>n&nbsp;</em>的数组，找出其中所有出现超过&nbsp;<code>⌊ n/3 ⌋</code>&nbsp;次的元素。</p>

<p><strong>说明: </strong>要求算法的时间复杂度为 O(n)，空间复杂度为 O(1)。</p>

<p><strong>示例&nbsp;1:</strong></p>

<pre><strong>输入:</strong> [3,2,3]
<strong>输出:</strong> [3]</pre>

<p><strong>示例 2:</strong></p>

<pre><strong>输入:</strong> [1,1,1,3,3,2,2,2]
<strong>输出:</strong> [1,2]</pre>
</div>

<strong>出现次数大于n/3的元素不会超过两个，先找出出现频率最高的两个元素，然后统计它们出现的次数</strong>

	class Solution 
	{
	public:
		void print(vector<int> &result)
		{
			for (auto &v : result)
			{
				cout << v << " ";
			}
			cout << endl;
		}
	
		vector<int> majorityElement(vector<int>& nums)
		{
			vector<int> result;
			if (nums.size() < 1)
			{
				return result;
			}
	
			int v1 = 0;
			int v2 = 0;
			
			int c1 = -1;
			int c2 = -1;
	
			for (int i = 0; i < nums.size(); i++)
			{
				if (c1 > 0 && nums[i] == v1)
				{
					c1++;
				}
				else if (c2 > 0 && nums[i] == v2)
				{
					c2++;
				}
				else if (c1 <= 0)
				{
					v1 = nums[i];
					c1 = 1;
				}
				else if (c2 <= 0)
				{
					v2 = nums[i];
					c2 = 1;
				}
				else
				{
					c1--;
					c2--;
				}
			}
	
			c1 = c2 = 0;
			for (int i = 0; i < nums.size(); i++)
			{
				if (nums[i] == v1)
				{
					c1++;
				}
				else if (nums[i] == v2)
				{
					c2++;
				}
			}
	
			if (c1 > nums.size() / 3)
			{
				result.push_back(v1);
			}
	
			if (c2 > nums.size() / 3)
			{
				result.push_back(v2);
			}
	
			return result;
		}
	};
